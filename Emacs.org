* Magit

https://magit.vc/manual/magit/

#+begin_src emacs-lisp

  (use-package magit
    :ensure t)

  ;; Custom functions for opening GitHub from magit
  (defun cem/parse-github-url (url)
    "convert a git remote location as a HTTP URL"
    (if (string-match "^http" url)
	url
      (replace-regexp-in-string "\\(.*\\)@\\(.*\\):\\(.*\\)\\(\\.git?\\)"
				"https://\\2/\\3"
				url)))

  (defun cem/magit-open-repo ()
    "open remote repo URL"
    (interactive)
    (let ((url (magit-get "remote" "origin" "url")))
      (progn
	(browse-url (cem/parse-github-url url))
	(message "Opening %s" url))))

  (add-hook 'magit-mode-hook
	    (lambda ()
	      (local-set-key (kbd "o") 'cem/magit-open-repo)))

  (message "successfully setup magit")

#+end_src
